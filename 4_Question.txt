#include<bits/stdc++.h>
using namespace std;

void ToggleBits(int &n){
    int temp = 1;
    while (temp <= n)
    {
        n ^= temp;
        temp <<= 1;
    }
}

int main(){
    int n;
    cin>>n;
    ToggleBits(n);
    cout<<n;
    return 0;
}
