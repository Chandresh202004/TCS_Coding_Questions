#include<iostream>
using namespace std;

bool prime(int n){
    if(n == 0 || n == 1) return 0;
    if(n == 2 || n == 3) return 1;

    for(int i = 2; i*i <= n; i++){
        if(n % i == 0) return 0;
    }

    return 1;
}

int main(){
    int num;
    cin>>num;

    if(num < 0) cout<<"please enter the positive number";
    else cout<<prime(num);

    return 0;
}
